function renderPeriodicTable(){
  if(periodicRendered) return;
  periodicRendered = true;
  var grid=document.getElementById('periodic-grid');
  grid.innerHTML='';
  layout.forEach(function(row){
    row.forEach(function(num){
      var cell=document.createElement('div');
      if(num===0){
        cell.className='element-cell empty-cell';
      }else if(num===999||num===998){
        cell.className='element-cell';
        cell.style.background='linear-gradient(135deg,rgba(94,92,230,0.6),rgba(130,80,220,0.6))';
        cell.style.border='1px solid rgba(94,92,230,0.4)';
        cell.innerHTML='<div style="font-size:15px;font-weight:800;color:#fff">'+(num===999?'57–71':'89–103')+'</div>';
      }else{
        var el=periodicElements.find(function(e){return e.num===num;});
        if(el){
          cell.className='element-cell '+el.category;
          cell.innerHTML=
            '<div class="element-number">'+el.num+'</div>'+
            '<div class="element-symbol">'+el.symbol+'</div>'+
            '<div class="element-name">'+el.name+'</div>'+
            '<div class="element-mass">'+el.mass+'</div>';
          cell.onclick=function(){showElementDetails(el);};
        }
      }
      grid.appendChild(cell);
    });
  });
}

// PDF LIBRARY
var pdfDoc=null,pageNum=1,scale=1.5,pageRendering=false,pageNumPending=null,rotation=0;
pdfjsLib.GlobalWorkerOptions.workerSrc='https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';

function loadPDFFile(event){
  var file=event.target.files[0];
  if(!file) return;
  var fileReader=new FileReader();
  fileReader.onload=function(){
    var typedarray=new Uint8Array(this.result);
    pdfjsLib.getDocument(typedarray).promise.then(function(pdf){
      pdfDoc=pdf;
      document.getElementById('page-count').textContent=pdf.numPages;
      var fsCount=document.getElementById('page-count-fs');
      if(fsCount) fsCount.textContent=pdf.numPages;
      pageNum=1;
      document.getElementById('pdf-viewer').style.display='block';
      document.getElementById('pdf-prev-arrow').style.display='flex';
      document.getElementById('pdf-next-arrow').style.display='flex';
      setTimeout(function(){ fitToWidth(); },100);
    });
  };
  fileReader.readAsArrayBuffer(file);
}
function renderPage(num){
  pageRendering=true;
  pdfDoc.getPage(num).then(function(page){
    var canvas=document.getElementById('pdf-canvas');
    var context=canvas.getContext('2d');
    var viewport=page.getViewport({scale:scale,rotation:rotation});
    canvas.height=viewport.height;
    canvas.width=viewport.width;
    page.render({canvasContext:context,viewport:viewport}).promise.then(function(){
      pageRendering=false;
      if(pageNumPending!==null){ renderPage(pageNumPending); pageNumPending=null; }
    });
    document.getElementById('page-num').textContent=num;
    var fsNum=document.getElementById('page-num-fs');
    if(fsNum) fsNum.textContent=num;
  });
}
function queueRenderPage(num){ if(pageRendering){ pageNumPending=num; }else{ renderPage(num); } }
function prevPage(){ if(pageNum<=1) return; pageNum--; queueRenderPage(pageNum); }
function nextPage(){ if(pageNum>=pdfDoc.numPages) return; pageNum++; queueRenderPage(pageNum); }
function firstPage(){ pageNum=1; queueRenderPage(pageNum); }
function lastPage(){ pageNum=pdfDoc.numPages; queueRenderPage(pageNum); }
function rotatePage(){ rotation=(rotation+90)%360; queueRenderPage(pageNum); }
function fitToWidth(){
  if(!pdfDoc) return;
  var container=document.getElementById('pdf-canvas-container');
  var containerWidth=container.clientWidth-60;
  pdfDoc.getPage(pageNum).then(function(page){
    var viewport=page.getViewport({scale:1,rotation:rotation});
    scale=containerWidth/viewport.width;
    queueRenderPage(pageNum);
  });
}
function toggleFullscreen(){
  var container=document.getElementById('pdf-canvas-container');
  if(!document.fullscreenElement){
    container.requestFullscreen().then(function(){
      document.getElementById('fullscreen-bar').style.display='flex';
      var fsCount=document.getElementById('page-count-fs');
      if(fsCount) fsCount.textContent=pdfDoc.numPages;
      setTimeout(function(){ fitToWidth(); },100);
    }).catch(function(err){ console.error('Fullscreen:',err); });
  }else{
    document.exitFullscreen();
  }
}
function exitFullscreenMode(){ if(document.fullscreenElement) document.exitFullscreen(); }
document.addEventListener('fullscreenchange',function(){
  if(!document.fullscreenElement) document.getElementById('fullscreen-bar').style.display='none';
});

// KEYBOARD SUPPORT FOR CALCULATOR
document.addEventListener('keydown', function(e){
  var calcSection = document.getElementById('calculator');
  if(!calcSection || calcSection.style.display === 'none') return;
  var k = e.key;
  if(k>='0'&&k<='9') append(k);
  else if(k==='+') append('+');
  else if(k==='-') append('-');
  else if(k==='*'||k==='x') append('*');
  else if(k==='/') { e.preventDefault(); append('/'); }
  else if(k==='.') append('.');
  else if(k==='%') append('%');
  else if(k==='Enter'||k==='=') calculate();
  else if(k==='Backspace') backspace();
  else if(k==='Escape') clearAll();
  else if(k==='(') append('(');
  else if(k===')') append(')');
});

var currentKey=null,currentData=null,currentSubject=null;
function showSubject(card){
  currentKey=card.getAttribute('data-subject');
  currentData=subjectsData[currentKey];
  if(!currentData) return;
  hideAll();
  var detail=document.getElementById('subject-detail');
  detail.style.display='block';
  requestAnimationFrame(function(){ detail.classList.add('section-reveal'); });
  document.getElementById('detail-title').textContent=currentData.title;
  renderSubjects();
}
function renderSubjects(){
  var grid=document.getElementById('subjects-grid');
  grid.innerHTML='';
  currentData.subjects.forEach(function(sub){
    var card=document.createElement('div');
    card.className='subject-card';
    card.innerHTML=
      '<span class="subject-icon">'+sub.icon+'</span>'+
      '<div class="subject-name">'+sub.name+'</div>'+
      '<div class="weight-badge">'+sub.weightage+'%</div>'+
      '<div style="margin-top:14px;color:var(--text-secondary);font-size:13px">'+sub.chapters.length+' Chapters</div>'+
      '<div class="progress-bar"><div class="progress-fill"></div></div>';
    card.onclick=function(){ showChapters(sub); };
    grid.appendChild(card);
  });
  document.getElementById('subjects-view').style.display='block';
  document.getElementById('chapters-view').style.display='none';
}
function showChapters(subject){
  currentSubject=subject;
  document.getElementById('subject-title').textContent=subject.name;
  var list=document.getElementById('chapters-list');
  list.innerHTML='';
  var bookmarks=JSON.parse(localStorage.getItem('quantum-bookmarks')||'[]');
  subject.chapters.forEach(function(ch,i){
    var li=document.createElement('li');
    var searchQuery=encodeURIComponent(currentData.title+' '+subject.name+' '+ch);
    var chapterId=currentKey+'-'+subject.name+'-'+ch;
    var isBookmarked=bookmarks.some(function(b){ return b.id===chapterId; });
    li.innerHTML=
      '<div style="display:flex;align-items:center;gap:12px;flex:1">'+
        '<button onclick="toggleBookmark(\''+chapterId.replace(/'/g,"\\'")+'\',\''+currentData.title.replace(/'/g,"\\'")+'\',\''+subject.name.replace(/'/g,"\\'")+'\',\''+ch.replace(/'/g,"\\'")+'\',event)" '+
          'style="background:none;border:none;font-size:20px;cursor:pointer;color:'+(isBookmarked?'var(--accent)':'var(--text-tertiary)')+';transition:all 0.2s" '+
          'class="bookmark-btn" data-id="'+chapterId+'">'+(isBookmarked?'🔖':'📑')+'</button>'+
        '<div style="flex:1">'+
          '<div style="font-size:15px;font-weight:600;margin-bottom:5px;color:var(--text-primary)">'+(i+1)+'. '+ch+'</div>'+
          '<a href="https://www.youtube.com/results?search_query='+searchQuery+'" target="_blank" '+
            'style="font-size:12px;color:var(--accent);text-decoration:none;font-weight:500" '+
            'onclick="event.stopPropagation()">📺 Watch on YouTube →</a>'+
        '</div>'+
      '</div>';
    list.appendChild(li);
  });
  document.getElementById('subjects-view').style.display='none';
  document.getElementById('chapters-view').style.display='block';
}
function toggleBookmark(id,classTitle,subjectName,chapterName,e){
  if(e) e.stopPropagation();
  var bookmarks=JSON.parse(localStorage.getItem('quantum-bookmarks')||'[]');
  var existingIndex=bookmarks.findIndex(function(b){ return b.id===id; });
  var isRemoving = existingIndex>-1;
  if(isRemoving){ bookmarks.splice(existingIndex,1); }
  else { bookmarks.push({id:id,class:classTitle,subject:subjectName,chapter:chapterName,timestamp:Date.now()}); }
  localStorage.setItem('quantum-bookmarks',JSON.stringify(bookmarks));
  var btn=document.querySelector('.bookmark-btn[data-id="'+id+'"]');
  if(btn){
    btn.textContent=isRemoving?'📑':'🔖';
    btn.style.color=isRemoving?'var(--text-tertiary)':'var(--accent)';
  }
  showToast(isRemoving ? '📑 Bookmark removed' : '🔖 Bookmarked: '+chapterName);
  renderBookmarks();
}
function renderBookmarks(){
  var bookmarks=JSON.parse(localStorage.getItem('quantum-bookmarks')||'[]');
  var section=document.getElementById('bookmarks-section');
  var list=document.getElementById('bookmarks-list');
  if(bookmarks.length===0){ section.style.display='none'; return; }
  section.style.display='block';
  list.innerHTML='';
  bookmarks.slice().reverse().forEach(function(b){
    var card=document.createElement('div');
    card.style.cssText='background:var(--glass-1);border:1px solid var(--glass-border);border-radius:var(--radius-md);padding:18px;display:flex;justify-content:space-between;align-items:center;transition:all 0.2s';
    card.onmouseenter=function(){ this.style.transform='translateX(5px)'; this.style.borderColor='rgba(10,132,255,0.3)'; };
    card.onmouseleave=function(){ this.style.transform='translateX(0)'; this.style.borderColor='var(--glass-border)'; };
    var searchQuery=encodeURIComponent(b.class+' '+b.subject+' '+b.chapter);
    card.innerHTML=
      '<div>'+
        '<div style="font-size:12px;color:var(--text-tertiary);margin-bottom:4px">'+b.class+' · '+b.subject+'</div>'+
        '<div style="font-size:15px;font-weight:600;margin-bottom:6px;color:var(--text-primary)">'+b.chapter+'</div>'+
        '<a href="https://www.youtube.com/results?search_query='+searchQuery+'" target="_blank" style="font-size:12px;color:var(--accent);text-decoration:none;font-weight:500">📺 Watch on YouTube →</a>'+
      '</div>'+
      '<button onclick="toggleBookmark(\''+b.id.replace(/'/g,"\\'")+'\',\''+b.class.replace(/'/g,"\\'")+'\',\''+b.subject.replace(/'/g,"\\'")+'\',\''+b.chapter.replace(/'/g,"\\'")+'\',event)" '+
        'style="background:none;border:none;font-size:24px;cursor:pointer;color:var(--accent)">🔖</button>';
    list.appendChild(card);
  });
}

  prevBtn.disabled=currentQuestionIndex===0;
  prevBtn.style.opacity=currentQuestionIndex===0?'0.4':'1';
  document.getElementById('next-btn').textContent=
    currentQuestionIndex===currentQuestions.length-1?'Finish ✓':'Next →';
}
function selectAnswer(index){
  userAnswers[currentQuestionIndex]=index;
  loadQuestion();
}
function prevQuestion(){
  if(currentQuestionIndex>0){ currentQuestionIndex--; loadQuestion(); }
}
function nextQuestion(){
  if(currentQuestionIndex<currentQuestions.length-1){
    currentQuestionIndex++; loadQuestion();
  }else{ finishMCQ(); }
}
function finishMCQ(){
  mcqScore=0;
  currentQuestions.forEach(function(q,i){
    if(userAnswers[i]===q.correct) mcqScore++;
  });
  document.getElementById('final-score').textContent=mcqScore+'/'+currentQuestions.length;
  var quizCard=document.querySelector('#mcq-quiz-view .quiz-card');
  if(quizCard) quizCard.style.display='none';
  document.getElementById('mcq-result').style.display='block';
  localStorage.setItem('quantum-quiz-score',mcqScore);
  document.getElementById('quiz-score').textContent=mcqScore;
}
function restartMCQ(){
  currentQuestionIndex=0;
  userAnswers=new Array(currentQuestions.length).fill(null);
  var quizCard=document.querySelector('#mcq-quiz-view .quiz-card');
  if(quizCard) quizCard.style.display='block';
  document.getElementById('mcq-result').style.display='none';
  loadQuestion();
}

(function initStats(){
  // Restore quiz score
  var savedScore = localStorage.getItem('quantum-quiz-score');
  if(savedScore) document.getElementById('quiz-score').textContent = savedScore;

  // Progress = bookmarks count as % of 30 possible
  var bookmarks = JSON.parse(localStorage.getItem('quantum-bookmarks')||'[]');
  var pct = Math.min(100, Math.round((bookmarks.length / 30) * 100));
  document.getElementById('total-progress').textContent = pct+'%';

  // Streak: check last visit date
  var today = new Date().toDateString();
  var lastVisit = localStorage.getItem('quantum-last-visit');
  var streak = parseInt(localStorage.getItem('quantum-streak')||'0');
  if(lastVisit){
    var yesterday = new Date(); yesterday.setDate(yesterday.getDate()-1);
    if(lastVisit === today){
      // same day, keep streak
    } else if(lastVisit === yesterday.toDateString()){
      streak++;
      localStorage.setItem('quantum-streak', streak);
    } else if(lastVisit !== today){
      streak = 1;
      localStorage.setItem('quantum-streak', streak);
    }
  } else {
    streak = 1;
    localStorage.setItem('quantum-streak', streak);
  }
  localStorage.setItem('quantum-last-visit', today);
  document.getElementById('study-streak').textContent = streak;
})();
